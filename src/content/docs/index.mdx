## Quick Start

```bash
npm install @vimee/core
```

```ts
import { createEditor, processKey } from "@vimee/core";

// Create an editor state — just data, no side effects
const editor = createEditor({ content: "Hello, vimee!" });

// Every operation is a pure function: state in, state out
const next = processKey(editor, { key: "x" });
// next.content === "ello, vimee!"
```
